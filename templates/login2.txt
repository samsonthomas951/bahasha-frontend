    <script>
        // Configuration
        const API_BASE_URL = 'http://localhost:5000/api/v1'; // Update this with your backend URL
        
        // DOM elements
        const loginForm = document.getElementById('login-form');
        const errorMessage = document.getElementById('error-message');
        const successMessage = document.getElementById('success-message');
        const loginButton = document.getElementById('login-button');
        const buttonText = document.getElementById('button-text');
        const buttonLoader = document.getElementById('button-loader');
        const debugInfo = document.getElementById('debug-info');
        const debugContent = document.getElementById('debug-content');
        
        // Debug function
        function showDebugInfo(data) {
            debugContent.innerHTML = `
                Request URL: ${API_BASE_URL}/auth/login<br>
                Request Data: ${JSON.stringify(data, null, 2)}<br>
                Identifier Length: ${data.identifier ? data.identifier.length : 'undefined'}<br>
                Password Length: ${data.password ? data.password.length : 'undefined'}<br>
                Identifier Trimmed: "${data.identifier}"<br>
                Password Present: ${data.password ? 'Yes' : 'No'}
            `;
            debugInfo.style.display = 'block';
        }
        
        // Handle login form submission
        loginForm.addEventListener('submit', async (e) => {
            e.preventDefault();
            
            // Clear messages
            errorMessage.classList.remove('show');
            successMessage.classList.remove('show');
            debugInfo.style.display = 'none';
            
            // Get form data with better validation
            const identifierField = document.getElementById('identifier');
            const passwordField = document.getElementById('password');
            
            const identifier = identifierField.value.trim();
            const password = passwordField.value; // Don't trim passwords as they might have intentional spaces
            
            // Client-side validation
            if (!identifier) {
                errorMessage.textContent = 'Please enter your email or username.';
                errorMessage.classList.add('show');
                identifierField.focus();
                return;
            }
            
            if (!password) {
                errorMessage.textContent = 'Please enter your password.';
                errorMessage.classList.add('show');
                passwordField.focus();
                return;
            }
            
            // Prepare request data - try different formats your backend might expect
            const requestData = {
                identifier: identifier,
                password: password,
                // Also include these in case your backend expects them
                email: identifier.includes('@') ? identifier : undefined,
                username: !identifier.includes('@') ? identifier : undefined
            };
            
            // Show debug info
            showDebugInfo(requestData);
            
            // Show loading state
            loginButton.disabled = true;
            buttonText.classList.add('hidden');
            buttonLoader.classList.remove('hidden');
            
            try {
                console.log('Sending login request:', requestData);
                
                const response = await fetch(`${API_BASE_URL}/auth/login`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify(requestData)
                });
                
                console.log('Response status:', response.status);
                console.log('Response headers:', Object.fromEntries(response.headers));
                
                const data = await response.json();
                console.log('Response data:', data);
                
                if (data.success) {
                    // Save auth data (only if not in artifact environment)
                    try {
                        const storage = window.localStorage || {};
                        storage.setItem?.('authToken', data.data.token);
                        storage.setItem?.('user', JSON.stringify(data.data.user));
                        storage.setItem?.('tokenExpiry', data.data.expires_at);
                    } catch (e) {
                        console.log('Storage not available, skipping auth save');
                    }
                    
                    // Show success message
                    successMessage.textContent = 'Login successful! Redirecting...';
                    successMessage.classList.add('show');
                    
                    // Redirect to dashboard (or show success in artifact)
                    setTimeout(() => {
                        try {
                            window.location.href = 'dashboard.html';
                        } catch (e) {
                            successMessage.textContent = 'Login successful! (Redirect disabled in preview)';
                        }
                    }, 1000);
                } else {
                    // Show error message with full details
                    const errorMsg = data.message || data.error || 'Login failed. Please try again.';
                    errorMessage.innerHTML = `
                        <strong>Login Failed:</strong> ${errorMsg}<br>
                        <small>Server Response: ${JSON.stringify(data, null, 2)}</small>
                    `;
                    errorMessage.classList.add('show');
                }
            } catch (error) {
                console.error('Login error:', error);
                errorMessage.innerHTML = `
                    <strong>Network Error:</strong> ${error.message}<br>
                    <small>Please check if your backend is running at ${API_BASE_URL}</small>
                `;
                errorMessage.classList.add('show');
            } finally {
                // Reset button state
                loginButton.disabled = false;
                buttonText.classList.remove('hidden');
                buttonLoader.classList.add('hidden');
            }
        });
        
        // Check if already logged in (only if storage is available)
        document.addEventListener('DOMContentLoaded', async () => {
            try {
                const storage = window.localStorage || {};
                const token = storage.getItem?.('authToken');
                if (token) {
                    try {
                        const response = await fetch(`${API_BASE_URL}/auth/verify`, {
                            headers: {
                                'Authorization': `Bearer ${token}`,
                            }
                        });
                        
                        const data = await response.json();
                        if (data.valid) {
                            // Already logged in, redirect to dashboard
                            try {
                                window.location.href = 'dashboard.html';
                            } catch (e) {
                                console.log('Redirect disabled in preview mode');
                            }
                        }
                    } catch (error) {
                        console.error('Verification error:', error);
                    }
                }
            } catch (e) {
                console.log('Storage not available, skipping auth check');
            }
        });
    </script>
